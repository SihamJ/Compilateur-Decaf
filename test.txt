class TripleFor {

    void main ()
    {
        int n;
        int s;
        int r;
        n = 10;
        s = 0;
        for i = 1, n
        {
            for j = 1, i
            {
                if(s > 7){break;}
                for k = 1, j
                {
                    s += 1;
                    continue;
                }
                if(s>3){ continue;}
                if(s == 8){ break;}
            }
        }
        r = n * (n + 1) * (n + 2) / 1 / 2 / 3;
        WriteInt (s);
        WriteString ("=");
        WriteInt (r);
        WriteString ("\n");
    }

}
